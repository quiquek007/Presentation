Installation

In a browser:

          <script src="lodash.js"></script>

In Node.js:

          $ npm i -g npm
          $ npm i --save lodash

Using npm:

          // Load the full build.
          var _ = require('lodash');

          // Load the core build.
          var _ = require('lodash/core');

          // Load method categories.
          var array = require('lodash/array');
          var object = require('lodash/fp/object');

_.range([start=0], end, [step=1])

Creates an array of numbers (positive and/or negative) progressing from start up to, but not including, end. 
            _.range(4);
            // => [0, 1, 2, 3]

            _.range(-4);
            // => [0, -1, -2, -3]

            _.range(1, 5);
            // => [1, 2, 3, 4]

            _.range(0, 20, 5);
            // => [0, 5, 10, 15]

            _.range(0, -4, -1);
            // => [0, -1, -2, -3]   
              pure JS
            Array.from({ length: 5 }, (v, k) => k);
            // [0, 1, 2, 3, 4]

_.drop(array, [n=1])

Creates a slice of array with n elements dropped from the beginning. 
              _.drop([1, 2, 3]);
              // => [2, 3]

              _.drop([1, 2, 3], 2);
              // => [3]

              _.drop([1, 2, 3], 5);
              // => []

              _.drop('string', 3);
              // => ['i', 'n', 'g']

_.dropRight(array, [n=1])

Creates a slice of array with n elements dropped from the end. 
          _.dropRight([1, 2, 3]);
          // => [1, 2]

          _.dropRight([1, 2, 3], 2);
          // => [1]

          _.dropRight([1, 2, 3], 5);
          // => []

          _.dropRight('string', 3);
          // => ['s', 't', 'r']

_.initial(array)

Gets all but the last element of array. 
          _.initial([1, 2, 3]);
          // => [1, 2]

          _.initial(['a', 'b', 'c']);
          // => ['a', 'b']

          _.initial('stringS');
          // => ['s', 't', 'r', 'i', 'n', 'g']

_.tail(array)

Gets all but the first element of array. 
          _.tail([1, 2, 3]);
          // => [2, 3]

          _.tail('abc');
          // => ['b', 'c']

_.take(array, [n=1]) & _.takeRight(array, [n=1])

Creates a slice of array with n elements taken from the beginning. 
          _.take([1, 2, 3]);  // => [1]

          _.take([1, 2, 3], 2); // => [1, 2]

          _.take([1, 2, 3], 5); // => [1, 2, 3]

          _.take('string', 2);  // => [ "s", "t" ]

          _.takeRight([1, 2, 3]); // => [3]

          _.takeRight([1, 2, 3], 2);  // => [2, 3]

          _.takeRight([1, 2, 3], 5);  // => [1, 2, 3]

          _.takeRight('string', 2); // => [ "n", "g" ]

_.floor(number, [precision=0]), _.round, _.ceil

Computes number rounded down to precision. 
          _.floor(4.006); // => 4

          _.floor(0.046, 2);  // => 0.04

          _.floor(4060, -2);  // => 4000

          _.round(4.006); // => 4

          _.round(4.006, 2);  // => 4.01

          _.round(4060, -2);  // => 4100

          _.ceil(4.006);  // => 5

          _.ceil(6.004, 2); // => 6.01

          _.ceil(6040, -2); // => 6100

_.*Case([string=''])

Converts string to * case. 
          _.camelCase('Foo Bar'); // => 'fooBar'

          _.kebabCase('Foo Bar'); // => 'foo-bar'

          _.lowerCase('--Foo-Bar--'); // => 'foo bar'

          _.snakeCase('fooBar'); // => 'foo_bar'

          _.startCase('--foo-bar--'); // => 'Foo Bar'

          _.upperCase('--foo-bar'); // => 'FOO BAR'

_.trim([string=''], [chars=whitespace])

Removes leading and trailing whitespace or specified characters from string. 
          _.trim('  abc  ');  // => 'abc'

          _.trim('-_-abc-_-', '_-');  // => 'abc'

    _.map(['  foo  ', '  bar  '], _.trim);  // => ['foo', 'bar']

          _.trimEnd('  abc  '); // => '  abc'

          _.trimEnd('-_-abc-_-', '_-'); // => '-_-abc'

          _.trimStart('  abc  '); // => 'abc  '

          _.trimStart('-_-abc-_-', '_-'); // => 'abc-_-'

_.pad([string=''], [length=0], [chars=' '])

Pads string on the left and right sides if it's shorter than length. Padding characters are truncated if they can't be evenly divided by length. 
          _.pad('abc', 8);  // => '  abc   '

          _.pad('abc', 8, '_-');  // => '_-abc_-_'

          _.pad('abc', 3);  // => 'abc'

          _.padEnd('abc', 6); // => 'abc   '

          _.padEnd('abc', 6, '_-'); // => 'abc_-_'

          _.padStart('abc', 6); // => '   abc'

          _.padStart('abc', 6, '_-'); // => '_-_abc'

          _.padStart('abc', 3); // => 'abc'

_.to*(value)

Converts value to a *<type>. 
          _.toArray({ 'a': 1, 'b': 2 });  // => [1, 2]

          _.toArray('abc'); // => ['a', 'b', 'c'];

          _.toInteger(3.2); // => 3

          _.toInteger(Infinity);  // => 1.7976931348623157e+308

          _.toInteger('3.2'); // => 3

          _.toString(null); // => ''

          _.toString([1, 2, 3]);  // => '1,2,3'
